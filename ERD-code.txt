Table users {
  id bigint [pk, increment]
  email varchar(255) [not null, unique]
  login varchar(255) [not null]
  name varchar(255)
  birthday date
}

Table ratings {
  id bigint [pk, increment]
  rating_name varchar(255)
}

Table genres {
  id bigint [pk, increment]
  genre_name varchar(255) [not null]
}

Table films {
  id bigint [pk, increment]
  name varchar(255) [not null]
  description varchar(200)
  release_date date
  duration integer
  rating_id bigint
}

Table film_genre {
  film_id bigint [not null]
  genre_id bigint [not null]
  indexes {
    (film_id, genre_id) [pk]
  }
}

Table user_like {
  film_id bigint [not null]
  user_id bigint [not null]
  indexes {
    (film_id, user_id) [pk]
  }
}

Table friendship {
  friend_from bigint [not null]
  friend_to bigint [not null]
  friendship_status varchar(20)
  indexes {
    (friend_from, friend_to) [pk]
  }
}

Ref: films.rating_id > ratings.id
Ref: film_genre.film_id > films.id
Ref: film_genre.genre_id > genres.id
Ref: user_like.film_id > films.id
Ref: user_like.user_id > users.id
Ref: friendship.friend_from > users.id
Ref: friendship.friend_to > users.id
